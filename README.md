Формат файла карты (она под названием tape):

{
  "tapeAuthor": "...",
  "trackname": "...",
  "trackAuthor" : "...",
  "tapeName" : "...",
  "version": 1,
  "audio": {
    "url": "/audio/track.mp3",
  },

  "timing": [
    { "t": 0,     "bpm": 175 },   // первый uninherited
    { "t": 12000, "bpm": 140 }    // можно ещё
  ],

  "notes": [
    { "t": 1234, "d": 0 },   // клик-нота на 0°
    { "t": 2345, "d": 90 },  // клик-нота на 90°
    { "t": 3456, "d": 180, "l": 400 }  // слайдер, 180°, длина 400 мс
  ]
}
